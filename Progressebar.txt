#!/bin/bash
###########################################################
#####################Progressebar##########################
###########################################################
$1 ===> source ; $2===> cible
(
 echo "10" ; sleep 1
 echo "20" ; sleep 1
 echo "50" ; sleep 1
 echo "75" ; sleep 1
 echo "100"; sleep 1
) |
zenity --progress \
  --title=$1 \
  --auto-close \
  --text=$2 \
  --percentage=0
  

if [ "$?" = -1 ] ; then
  zenity --error \
    --text="opération annulée."	
fi
}
